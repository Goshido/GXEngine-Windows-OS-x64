#version 330 core


#define COLOR_SLOT	0


smooth in vec3 g_locationWorld;

uniform samplerCube environmentSampler;

out layout ( location = COLOR_SLOT ) vec3 fragmentColor;


void main ()
{
	vec3 v = vec3 ( g_locationWorld.x, -g_locationWorld.y, g_locationWorld.z );
	fragmentColor = texture ( environmentSampler, v ).rgb;
}
